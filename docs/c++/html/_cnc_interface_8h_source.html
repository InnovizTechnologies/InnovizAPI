488 template<typename T>
489 Result ICnc::GetParameterValue(const DataPoint* parameter_datapoint, T& parameter_value, uint32_t policy)
490 {
491 RESULT_START
492 /* Get type name */
493 std::string typeName(typeid(T).name());
494
495 /* Validate T size corresponds with the DP size */
496 if (sizeof(T) != parameter_datapoint->size)
497 {
498 std::string error_msg{ "Type mismatch asked datapoint type is " + parameter_datapoint->type + " used type is " + typeName };
499 result.error_code = ERROR_CODE_GENERAL;
500 result.error_message = error_msg;
501 throw std::runtime_error(error_msg);
502 }
503 /* Scalar type non-pointer */
504 else
505 {
506 size_t paramTypeLen = sizeof(T);
507 uint8_t* paramByteArray = reinterpret_cast<uint8_t*>(&parameter_value);
508 result = GetParameterById(parameter_datapoint->id, paramTypeLen, paramByteArray, policy);
509 if (result.error_code != ERROR_CODE_OK)
510 return result;
511 }
512 RESULT_END
513 }

515 template<>
516 inline
517 Result ICnc::GetParameterValue<std::string>(const DataPoint* parameter_datapoint, std::string& parameter_value, uint32_t policy)
518 {
519 RESULT_START
520 size_t strLen = parameter_datapoint->size;
521 uint8_t* strParamArray = new uint8_t[strLen + 1]{ 0 };
522 result = GetParameterById(parameter_datapoint->id, strLen, strParamArray, policy);
523 strParamArray[strLen] = '\0';
524 if (result.error_code != ERROR_CODE_OK)
525 return result;
526 std::string strParam = (char*)strParamArray;
527 delete[] strParamArray;
528 parameter_value = strParam;
529 RESULT_END
530 }

532 template<>
533 inline
534 Result ICnc::GetParameterValue<bool>(const DataPoint* parameter_datapoint, bool& parameter_value, uint32_t policy)
535 {
536 RESULT_START
537 if ((parameter_datapoint->rows * parameter_datapoint->columns) == 1)
538 {
539 size_t paramTypeLen = parameter_datapoint->size;
540 uint8_t* paramByteArray = new uint8_t[paramTypeLen];
541 result = GetParameterById(parameter_datapoint->id, paramTypeLen, paramByteArray, policy);
542 if (result.error_code != ERROR_CODE_OK)
543 return result;
544 parameter_value = (bool)(*paramByteArray);
545 delete[] paramByteArray;
546 }
547 else
548 {
549 std::string error_msg{ "GetParameterValue can return value of one item only" };
550 result.error_code = ERROR_CODE_GENERAL;
551 result.error_message = error_msg;
552 throw std::runtime_error(error_msg);
553 }
554 RESULT_END
555 }
