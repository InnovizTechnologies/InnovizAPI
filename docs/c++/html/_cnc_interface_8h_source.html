487  template<typename T>
488  Result ICnc::GetParameterValue(const DataPoint* parameter_datapoint, T& parameter_value, uint32_t policy)
489  {
490  RESULT_START
491  /* Get type name */
492  std::string typeName(typeid(T).name());
493 
494  /* Validate T size corresponds with the DP size */
495  if (sizeof(T) != parameter_datapoint->size)
496  {
497  std::string error_msg{ "Type mismatch asked datapoint type is " + parameter_datapoint->type + " used type is " + typeName };
498  result.error_code = ERROR_CODE_GENERAL;
499  result.error_message = error_msg;
500  throw std::runtime_error(error_msg);
501  }
502  /* Scalar type non-pointer */
503  else
504  {
505  size_t paramTypeLen = sizeof(T);
506  uint8_t* paramByteArray = reinterpret_cast<uint8_t*>(&parameter_value);
507  result = GetParameterById(parameter_datapoint->id, paramTypeLen, paramByteArray, policy);
508  if (result.error_code != ERROR_CODE_OK)
509  return result;
510  }
511  RESULT_END
512  }

517  {
518  RESULT_START
