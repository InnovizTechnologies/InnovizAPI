487 template<typename T>
488 Result ICnc::GetParameterValue(const DataPoint* parameter_datapoint, T& parameter_value, uint32_t policy)
489 {
490 RESULT_START
491 /* Get type name */
492 std::string typeName(typeid(T).name());
493
494 /* Validate T size corresponds with the DP size */
495 if (sizeof(T) != parameter_datapoint->size)
496 {
497 std::string error_msg{ "Type mismatch asked datapoint type is " + parameter_datapoint->type + " used type is " + typeName };
498 result.error_code = ERROR_CODE_GENERAL;
499 result.error_message = error_msg;
500 throw std::runtime_error(error_msg);
501 }
502 /* Scalar type non-pointer */
503 else
504 {
505 size_t paramTypeLen = sizeof(T);
506 uint8_t* paramByteArray = reinterpret_cast<uint8_t*>(&parameter_value);
507 result = GetParameterById(parameter_datapoint->id, paramTypeLen, paramByteArray, policy);
508 if (result.error_code != ERROR_CODE_OK)
509 return result;
510 }
511 RESULT_END
512 }

514 template<>
515 inline
516 Result ICnc::GetParameterValue<std::string>(const DataPoint* parameter_datapoint, std::string& parameter_value, uint32_t policy)
517 {
518 RESULT_START
519 size_t strLen = parameter_datapoint->size;
520 uint8_t* strParamArray = new uint8_t[strLen + 1]{ 0 };
521 result = GetParameterById(parameter_datapoint->id, strLen, strParamArray, policy);
522 strParamArray[strLen] = '\0';
523 if (result.error_code != ERROR_CODE_OK)
524 return result;
525 std::string strParam = (char*)strParamArray;
526 delete[] strParamArray;
527 parameter_value = strParam;
528 RESULT_END
529 }

532 inline
533 Result ICnc::GetParameterValue<bool>(const DataPoint* parameter_datapoint, bool& parameter_value, uint32_t policy)
534 {
535 RESULT_START
536 if ((parameter_datapoint->rows * parameter_datapoint->columns) == 1)
537 {
538 size_t paramTypeLen = parameter_datapoint->size;
539 uint8_t* paramByteArray = new uint8_t[paramTypeLen];
540 result = GetParameterById(parameter_datapoint->id, paramTypeLen, paramByteArray, policy);
541 if (result.error_code != ERROR_CODE_OK)
542 return result;
543 parameter_value = (bool)(*paramByteArray);
544 delete[] paramByteArray;
545 }
546 else
547 {
548 std::string error_msg{ "GetParameterValue can return value of one item only" };
549 result.error_code = ERROR_CODE_GENERAL;
550 result.error_message = error_msg;
551 throw std::runtime_error(error_msg);
552 }
553 RESULT_END
