90 CSampleFrameMetaDataTLV(uint32_t _frame_number, uint8_t _scan_mode, uint8_t _system_mode, uint8_t _system_submode, uint32_t _timestamp_internal,
91 uint32_t _timestamp_utc_sec, uint32_t _timestamp_utc_micro, uint32_t _fw_version, uint32_t _hw_version, uint8_t* p_lidar_serial,
92 uint16_t _device_type, uint8_t _active_lrfs, uint8_t _macro_pixel_shape, uint8_t* _rows_in_lrf, uint16_t* _cols_in_lrf, uint32_t _total_number_of_points,
93 ReflectionMatrix* _R_i, vector3* _d_i, ChannelNormal* _v_i_k)
94 {
95 memset(this, 0, sizeof(CSampleFrameMetaDataTLV));
96 header.type = POINT_CLOUD_CSAMPLE_METADATA;
97 header.length = sizeof(CSampleFrameMetaDataTLV) - sizeof(BlockTLV);
98 frame_number = _frame_number;
99 scan_mode = _scan_mode;
100 system_mode = _system_mode;
101 system_submode = _system_submode;
102 timestamp_internal = _timestamp_internal;
103 timestamp_utc_secs = _timestamp_utc_sec;
104 timestamp_utc_micro = _timestamp_utc_micro;
105 fw_version = _fw_version;
106 hw_version = _hw_version;
107 memcpy(lidar_serial_number, p_lidar_serial, INVZ4_CSAMPLE_LIDAR_SERIAL_LENGTH);
108 device_type = _device_type;
109 active_lrfs = _active_lrfs;
110 macro_pixel_shape = _macro_pixel_shape; /* (4bits,4bits) */
111 memcpy(rows_in_lrf,_rows_in_lrf, DEVICE_NUM_OF_LRFS * sizeof(uint8_t));
112 memcpy(cols_in_lrf, _cols_in_lrf, DEVICE_NUM_OF_LRFS * sizeof(uint16_t));
113 total_number_of_points = _total_number_of_points;
114 memcpy(R_i, _R_i, DEVICE_NUM_OF_LRFS * sizeof(ReflectionMatrix));
115 memcpy(d_i, _d_i, DEVICE_NUM_OF_LRFS * sizeof(vector3));
116 memcpy(v_i_k, _v_i_k, DEVICE_NUM_OF_LRFS * sizeof(ChannelNormal));
117 };
