375  template<typename T>
376  void GetArrayValue(DataPoint *dp, size_t arr_size, T *arr)
377  {
378  // Validate array size
379  if (arr_size == 0 || arr_size != (dp->size / sizeof(T))) {
380  std::string error{ "Bad array size. This datapoint is of size: " };
381  error.append(std::to_string(dp->size / sizeof(T)));
382  error.append(" And you were asking for: ");
383  error.append(std::to_string(arr_size));
384  throw std::runtime_error(error);
385  }
386 
387  // Validate T size corresponds with the DP size
388  if (sizeof(T) != (dp->size / arr_size)) {
389  std::string error{ "Bad type. This datapoint is of type: " };
390  error.append(dp->type);
391  error.append(" And you were asking for: ");
392  error.append(typeid(T).name());
393  throw std::runtime_error(error);
394  }
395 
396  // Validate DP initalization
397  if (dp == NULL || dp == nullptr) {
398  throw std::runtime_error("Datapoint is not initialized.");
399  }
400 
401  // Get value byte array style and put in the requested array
402  GetByteArray(dp, (byte*)arr);
403  }

415  template<typename T>
416  T GetValue(DataPoint *dp)
417  {
418  // Validate T size corresponds with the DP size
419  if (sizeof(T) != dp->size) {
420  std::string error{ "Bad type. This datapoint is of type: " };
421  error.append(dp->type);
422  error.append(" And you were asking for: ");
423  error.append(typeid(T).name());
424  throw std::runtime_error(error);
425  }
426 
427  // Validate DP initalization
428  if (dp == NULL || dp == nullptr) {
429  throw std::runtime_error("Datapoint is not initialized.");
430  }
431 
432  // Get value byte array style
433  byte *byte_arr = new byte[sizeof(T)];
434  GetByteArray(dp, byte_arr);
435 
436  // Cast value to T type
437  T value = *(reinterpret_cast<T*>(byte_arr));
438 
439  // Free dynamic allocation and return
440  delete[] byte_arr;
441  return value;
442  }
