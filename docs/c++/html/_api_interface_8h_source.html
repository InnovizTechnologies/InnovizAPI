399  template<typename T>
400  void GetArrayValue(DataPoint *dp, size_t arr_size, T *arr)
401  {
402  // Validate array size
403  if (arr_size == 0 || arr_size != (dp->size / sizeof(T))) {
404  std::string error{ "Bad array size. This datapoint is of size: " };
405  error.append(std::to_string(dp->size / sizeof(T)));
406  error.append(" And you were asking for: ");
407  error.append(std::to_string(arr_size));
408  throw std::runtime_error(error);
409  }
410 
411  // Validate T size corresponds with the DP size
412  if (sizeof(T) != (dp->size / arr_size)) {
413  std::string error{ "Bad type. This datapoint is of type: " };
414  error.append(dp->type);
415  error.append(" And you were asking for: ");
416  error.append(typeid(T).name());
417  throw std::runtime_error(error);
418  }
419 
420  // Validate DP initalization
421  if (dp == NULL || dp == nullptr) {
422  throw std::runtime_error("Datapoint is not initialized.");
423  }
424 
425  // Get value byte array style and put in the requested array
426  GetByteArray(dp, (byte*)arr);
427  }

439  template<typename T>
440  T GetValue(DataPoint *dp)
441  {
442  // Validate T size corresponds with the DP size
443  if (sizeof(T) != dp->size) {
444  std::string error{ "Bad type. This datapoint is of type: " };
445  error.append(dp->type);
446  error.append(" And you were asking for: ");
447  error.append(typeid(T).name());
448  throw std::runtime_error(error);
449  }
450 
451  // Validate DP initalization
452  if (dp == NULL || dp == nullptr) {
453  throw std::runtime_error("Datapoint is not initialized.");
454  }
455 
456  // Get value byte array style
457  byte *byte_arr = new byte[sizeof(T)];
458  GetByteArray(dp, byte_arr);
459 
460  // Cast value to T type
461  T value = *(reinterpret_cast<T*>(byte_arr));
462 
463  // Free dynamic allocation and return
464  delete[] byte_arr;
465  return value;
466  }
